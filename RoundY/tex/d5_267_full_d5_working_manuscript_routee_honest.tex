\section{Graph-theoretic setup}

\begin{definition}[directed torus]
For \(m>2\), write
\[
D_5(m)=\operatorname{Cay}((\Z/m\Z)^5,\{e_0,e_1,e_2,e_3,e_4\}).
\]
Its vertices are the points of \((\Z/m\Z)^5\), and from each vertex \(x\) there
are exactly five directed outgoing arcs
\[
x\to x+e_0,\ \dots,\ x\to x+e_4.
\]
\end{definition}

\begin{definition}[Hamilton decomposition]
A \emph{Hamilton decomposition} of \(D_5(m)\) is a partition of the full
directed arc set into five pairwise arc-disjoint directed Hamilton cycles.
\end{definition}

\section{Three languages and a methodological rule}

\begin{remark}[support language, grid language, odometer language]
Three languages appear repeatedly and should not be confused.
\begin{enumerate}[label=\textup{(\arabic*)},leftmargin=2.8em]
\item \textbf{Support language.}
      This records only which local families are present.
      It is a pruning language.
\item \textbf{Grid language.}
      This records actual endpoint words, equivalently the grid cell
      \(\sig{i}{j}\).
      It is the main language of shape-side row theorems.
\item \textbf{Odometer language.}
      This records the explicit post-entry coordinates
      \((q,w,u,\Theta)\).
      It is the main language after the canonical entry theorem.
\end{enumerate}
The key rule is:
\emph{support is a pruning invariant, not a classification invariant.}
So support can help isolate a theorem target, but it cannot by itself separate
all rows that the grid language can separate.
\end{remark}

\begin{theorem}[surfaced-bundle closure of the \(\sig23\) side]
\label{thm:sigma23-closed}
For every modulus \(m>2\), the actual second step of the surviving immediate
\(R1\)-family out of
\[
\sig21=[4,1,4]/[1,2,2]
\]
agrees with the \(\sig22=[4,1,4]/[2,1,2]\) prototype and never with the
\(\sig23=[4,1,4]/[2,2,1]\) prototype.
Equivalently, for the tiny-repair-normalized surviving immediate \(R1\)-image
\(\RhoR\) one has
\[
\Delta_R(\RhoR)=\{1,2\}.
\]
In particular the left-strip burden inside \(T1\) is closed.
\end{theorem}

\begin{proof}
Fix a modulus \(m>2\), color \(0\), and the standard normalization
\(w_0=s_0=0\).
Write \(\partial_i\) for the coordinate step that adds \(+1\) in direction
\(i\) modulo \(m\).
In the surfaced builder, the common right \(R1\)-source family for
\(\sig21,\sig22,\sig23\) is exactly
\[
S_R=\{x:\ \mathrm{layer}(x)=1,\ q(x)=m-1,\ w(x)=0,\ s(x)\neq 0\}.
\]
Thus all three rows are compared on the same source family.

For \(\sig21\), the right word is \([1,2,2]\), so the first two right-side
steps on \(S_R\) are
\[
x\mapsto \partial_1(x)\mapsto \partial_2\partial_1(x).
\]
For \(\sig22\), the right word is \([2,1,2]\), so the first two right-side
steps are
\[
x\mapsto \partial_2(x)\mapsto \partial_1\partial_2(x).
\]
For \(\sig23\), the right word is \([2,2,1]\), so the first two right-side
steps are
\[
x\mapsto \partial_2(x)\mapsto \partial_2^2(x).
\]

The surfaced common module defines each \(\partial_i\) as coordinatewise
\(+1\) in a single axis modulo \(m\).  Therefore the coordinate steps
commute:
\[
\partial_2\partial_1=\partial_1\partial_2.
\]
Thus the actual second step of \(\sig21\) on the surviving \(R1\)-family agrees
exactly with the \(\sig22\) prototype.

On the other hand, every \(x\in S_R\) satisfies \((q(x),w(x))=(m-1,0)\).
Therefore
\[
\partial_2\partial_1(x):(q,w)=(0,1),\qquad
\partial_2^2(x):(q,w)=(m-1,2).
\]
These are distinct for every \(m>2\).  Hence the actual second step of
\(\sig21\) does not realize \(\sig23\).
The description of \(\Delta_R\) now follows from
Proposition~\ref{prop:left-strip-case-split}.
\end{proof}

\begin{theorem}[actual color-$3$ \(P_0\) return formula]
\label{thm:M5-color3-R}
For every modulus \(m>2\), let
\[
\Pzero=\Bigl\{x\in(\Z/m\Z)^5:\ x_0+x_1+x_2+x_3+x_4=0\Bigr\},
\qquad
R_3^\star=(F_3^\star)^m\vert_{\Pzero},
\]
where \(F_3^\star\) is the color-$3$ torus map induced by \(\SelStar\).
Then for every \(x=(x_0,x_1,x_2,x_3,x_4)\in \Pzero\),
\[
R_3^\star(x)=x+e_4+e_2+e_{d(x)}-3e_3,
\]
where \(d(x)\) is given by:
\begin{enumerate}[label=\textup{(\arabic*)},leftmargin=2.8em]
\item if \(x_2=m-2\), \(x_3=m-1\), and \(x_4=m-2\), then \(d(x)=0\);
\item else if \(x_2=m-2\), \(x_0=m-1\), and \(x_3\neq m-1\), then \(d(x)=4\);
\item else if \(x_2=m-2\), then \(d(x)=1\);
\item else if \(x_1=m-1\) and \(x_4=m-2\), then \(d(x)=0\);
\item else \(d(x)=3\).
\end{enumerate}
In particular,
\[
x_2(R_3^\star(x))=x_2+1.
\]
\end{theorem}

\begin{proof}
Over the \(m\) steps defining \(R_3^\star\), the exact sum slice runs once
through \(0,1,2,3,\dots,m-1\).
For color \(3\), the contributions on slices \(0,1,3,\ge 4\) are fixed:
\[
\Sigma=0:\ 4,\qquad
\Sigma=1:\ 2,\qquad
\Sigma=3:\ 3,\qquad
\Sigma\ge 4:\ 3.
\]
So the only nontrivial step is the unique \(\Sigma=2\) step.
At that moment the state is \(x+e_4+e_2\), so the active-index test for the
slice-\(2\) rule reads
\[
\begin{aligned}
0&\iff x_1=m-1, &\qquad 1&\iff x_2=m-2,\\
2&\iff x_3=m-1, &\qquad 3&\iff x_4=m-2,\\
4&\iff x_0=m-1. &&
\end{aligned}
\]
Reading the explicit \(q^{(2,\star)}\) table for color \(3\) gives exactly the
five cases above.
Adding this \(\Sigma=2\) contribution to the fixed contributions from the other
slices yields the formula for \(R_3^\star(x)\).
The \(x_2\)-clock statement is immediate because every case contributes exactly
one \(e_2\).
\end{proof}

\begin{theorem}[actual color-$3$ section identification]
\label{thm:M5-color3-ident}
Fix \(m\ge 9\) and write
\[
\Ssec=\{x\in\Pzero:\ x_2=m-2\},
\qquad
T_m=(R_3^\star)^m\vert_{\Ssec}.
\]
On \(\Ssec\), use coordinates
\[
(a,b,e)=(x_0,x_1,x_4),
\qquad
c=x_3=2-a-b-e \pmod m.
\]
Then \(T_m\) is exactly the explicit five-branch model \(\Tsharp\):
\begin{enumerate}[label=\textup{(\arabic*)},leftmargin=2.8em]
\item if \(a=m-1\), \(c\neq m-1\), and \((b\neq m-1\ \text{or}\ e=m-3)\), then
      \((a,b,e)\mapsto(a,b,e+1)\);
\item if \(b=m-2\), \(e\neq m-2\), and \((a\neq m-1\ \text{or}\ e=6)\), then
      \((a,b,e)\mapsto(a+1,b+1,e)\);
\item if \(e=m-2\) and \(c=m-1\), then
      \((a,b,e)\mapsto(a+1,b,e)\);
\item if \(a=m-1\), \(b=m-1\), and \(e\notin\{5,m-3\}\), then
      \((a,b,e)\mapsto(a+1,b,e+1)\);
\item otherwise
      \((a,b,e)\mapsto(a,b+1,e)\).
\end{enumerate}
\end{theorem}

\begin{proof}
Take \(x=(a,b,m-2,c,e)\in \Ssec\).
By Theorem~\ref{thm:M5-color3-R}, the first \(R_3^\star\) step falls into one
of three initial classes.

If \(e=m-2\) and \(c=m-1\), the initial step is the special \(d(x)=0\) case, so
after one \(R_3^\star\) step the state is
\[
(a+1,b,m-1,c-3,m-1).
\]
On the remaining \(m-1\) steps, the condition \(x_4=m-2\) never occurs, so no
later special step can occur.  Therefore the orbit returns to \(\Ssec\) at
\[
(a+1,b,e),
\]
which is branch \textup{(3)}.

If \(a=m-1\) and \(c\neq m-1\), the initial step is the special \(d(x)=4\)
case, so after one \(R_3^\star\) step the state is
\[
(a,b,m-1,c-3,e+2).
\]
On the remaining \(m-1\) steps, the only possible nongeneric event is the
later-special case \(x_1=m-1\) and \(x_4=m-2\).  Here \(x_1=b\) is fixed, while
\(x_4\) starts at \(e+2\) and increases by \(1\) each time.  So a later-special
step occurs exactly when \(b=m-1\) and \(e\neq m-3\).  Inside the present
initial class, the condition \(c\neq m-1\) excludes exactly the exceptional
value \(e=5\).  Hence:
\begin{itemize}[leftmargin=2.4em]
\item if \(b\neq m-1\) or \(e=m-3\), no later-special step occurs and the final
      section image is \((a,b,e+1)\), giving branch \textup{(1)};
\item if \(b=m-1\) and \(e\notin\{5,m-3\}\), one later-special step occurs and
      the final section image is \((a+1,b,e+1)\), giving branch \textup{(4)}.
\end{itemize}

In every remaining case, the initial step is the generic \(d(x)=1\) case, so
after one \(R_3^\star\) step the state is
\[
(a,b+1,m-1,c-3,e+1).
\]
Again the only possible nongeneric event later is \(x_1=m-1\) and \(x_4=m-2\).
This happens exactly when \(b=m-2\) and \(e\neq m-2\).  Within the present
initial class, the failure of the previous \(a=m-1,c\neq m-1\) branch is
equivalent, when \(b=m-2\), to \((a\neq m-1\ \text{or}\ e=6)\).  Therefore:
\begin{itemize}[leftmargin=2.4em]
\item if \(b=m-2\), \(e\neq m-2\), and \((a\neq m-1\ \text{or}\ e=6)\), one
      later-special step occurs and the final section image is
      \((a+1,b+1,e)\), giving branch \textup{(2)};
\item otherwise no later-special step occurs and the final section image is
      \((a,b+1,e)\), giving branch \textup{(5)}.
\end{itemize}

These cases exhaust \(\Ssec\), so \(T_m=\Tsharp\).
\end{proof}

\section{Final upgrade and main theorem}

The endgame is now the corrected generic odd-$m$, $d=5$ upgrade step.
Compared with the older formulation, the corrected criterion does \emph{not}
ask for a single common selector family and does \emph{not} use the false
future-word separation target for the current event map.
Instead it asks for:
\begin{enumerate}[label=\textup{(\arabic*)},leftmargin=2.8em]
\item the best-seed channel classification on the front end;
\item the theorem-side anchor/globalization package;
\item a five-color graph-side assembly package on the full torus.
\end{enumerate}

\begin{theorem}[corrected generic odd-$m$, $d=5$ upgrade criterion]
\label{thm:generic-upgrade}
Let $m\ge 5$ be odd.
Assume:
\begin{enumerate}[label=\textup{(G\arabic*)},leftmargin=3.0em]
\item the front-end channel classification of
      Corollary~\ref{cor:channel-classification};
\item the theorem-side anchor package of Theorem~\ref{thm:M23} together with
      the globalization theorem~\ref{thm:M6-globalization};
\item the five-color graph-side assembly input of
      Background~\ref{bg:five-color-assembly}.
\end{enumerate}
Then the directed torus $D_5(m)$ admits a decomposition into five arc-disjoint
Hamilton cycles.
\end{theorem}

\begin{proof}
By (G3), the five color maps determine five selected arc sets whose outgoing
arcs at each vertex are exactly the five torus outgoing arcs and whose incoming
arcs are unique colorwise.  Hence the five global maps are torus permutations
with pairwise disjoint arc sets covering the full directed arc set.
By the same input, each of those permutations is a Hamilton cycle.
Therefore the five arc sets form a Hamilton decomposition of $D_5(m)$.
The role of (G1) and (G2) is not additional combinatorics at this last step;
it is to identify the selector package and its Hamiltonicity with the actual
odd-$m$, $d=5$ branch selected by the front-end and theorem-side packets.
\end{proof}

\begin{proof}[Proof of Theorem~\ref{thm:conditional-full}]
Fix an odd modulus $m\ge 5$.
The front-end theorem block closes the seed search and leaves a unique live
channel; Corollary~\ref{cor:channel-classification} records this as the
best-seed one-corner channel.
The post-entry dynamics on that channel are then governed by the explicit
odometer spine of Theorem~\ref{thm:odometer-spine}.
On the theorem side, Theorem~\ref{thm:M23} supplies the exact anchor package
and Theorem~\ref{thm:M6-globalization} supplies the odd-$m$ globalization
conclusion.
On the graph side, Background~\ref{bg:five-color-assembly} supplies the needed
five-color Hamilton package, while Theorems~\ref{thm:M5-color3} and
\ref{thm:M5-color4} identify two of those colors with the closed Sel\(^{\star}\)
branches constructed explicitly in the graph-side packets.
Therefore all hypotheses of Theorem~\ref{thm:generic-upgrade} are satisfied,
and the directed torus $D_5(m)$ admits a decomposition into five arc-disjoint
directed Hamilton cycles.
\end{proof}
